Implémente la détection automatique de conflits d'intérêts pour respecter les obligations déontologiques de l'Ordre des avocats tunisien.

PRÉREQUIS : T17 livré (table case_opponents avec FK vers opponents). T3 livré (table clients avec tax_id).

ÉTAPE 1 — SCHÉMA
═══════════════════════════════════════════════════════

Table conflict_checks :

CREATE TABLE conflict_checks (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  case_id uuid NOT NULL REFERENCES cases(id) ON DELETE CASCADE,
  conflict_type enum NOT NULL, -- 'opponent_is_client', 'client_is_opponent_elsewhere', 'shared_party'
  conflicting_entity_type enum NOT NULL, -- 'client', 'opponent'
  conflicting_entity_id uuid NOT NULL,
  matched_on enum NOT NULL, -- 'tax_id_exact', 'name_fuzzy', 'name_exact'
  match_score numeric, -- 0-1 pour fuzzy match
  detected_at timestamptz DEFAULT now(),
  resolved boolean DEFAULT false,
  resolved_at timestamptz,
  resolved_by uuid REFERENCES users(id),
  resolution_justification text, -- justification écrite de l'avocat
  created_at timestamptz DEFAULT now()
);

CREATE INDEX idx_conflict_checks_case ON conflict_checks(case_id);
CREATE INDEX idx_conflict_checks_unresolved ON conflict_checks(case_id) WHERE resolved = false;

ÉTAPE 2 — SERVICE DE DÉTECTION
═══════════════════════════════════════════════════════

Crée src/services/conflictDetection.ts :

class ConflictDetectionService {
  static async checkForCase(caseId: string): Promise<ConflictCheck[]> {
    const case = await fetchCase(caseId);
    const opponents = await fetchCaseOpponents(caseId);
    const conflicts: ConflictCheck[] = [];

    // Vérification 1 : chaque adversaire de ce dossier est-il client ailleurs ?
    for (const opponent of opponents) {
      const matchingClients = await findMatchingClients(opponent);
      for (const client of matchingClients) {
        conflicts.push({
          conflict_type: 'opponent_is_client',
          conflicting_entity_type: 'client',
          conflicting_entity_id: client.id,
          matched_on: client.matchedOn,
          match_score: client.score
        });
      }
    }

    // Vérification 2 : le client de ce dossier est-il adversaire dans un autre dossier ?
    const clientAsOpponent = await findClientAsOpponentInOtherCases(case.client_id, caseId);
    for (const otherCaseOpponent of clientAsOpponent) {
      conflicts.push({
        conflict_type: 'client_is_opponent_elsewhere',
        conflicting_entity_type: 'opponent',
        conflicting_entity_id: otherCaseOpponent.id,
        // ...
      });
    }

    return conflicts;
  }
}

ALGORITHME findMatchingClients :
1. Si opponent.tax_id existe : SELECT * FROM clients WHERE tax_id = opponent.tax_id → matched_on = 'tax_id_exact', score 1.0
2. SELECT * FROM clients WHERE 
     LOWER(name_ar) = LOWER(opponent.name_ar) 
     OR LOWER(name_fr) = LOWER(opponent.name_fr)
   → matched_on = 'name_exact', score 0.9
3. Fuzzy match avec pg_trgm ou similar :
   SELECT * FROM clients 
   WHERE similarity(name_ar, opponent.name_ar) > 0.7 
   OR similarity(name_fr, opponent.name_fr) > 0.7
   → matched_on = 'name_fuzzy', score = similarity value

Seuil de match : 0.7 minimum pour fuzzy. En dessous, ignoré.
